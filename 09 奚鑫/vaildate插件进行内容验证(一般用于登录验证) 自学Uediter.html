---validate插件
                 链接: https://www.runoob.com/jquery/jquery-plugin-validate.html

一  对表单元素设置validate方法,表示要对指定表单元素进行验证操作
      - rules 用于指定匹配的规则
                 - 使用某个表单元素的name作为属性名使用,后续代码可以设置指定的规则
      - messages 用于指定每种规则对应的错误信息(默认的信息为英文形式,可以自定义为任意的内容,使用方便)

$form.validate({
      rules : {
          email_name : {
                required : true,              // 指定是否为必选项 : required
                email : true,                   // 指定匹配规则
                maxlength : 20,             // 最大长度
                minlength : 6                // 最小长度
           },
        
          password_name : {
                required : true,
                psw : true,
                equalTo : '#repsw'        // 指定当前元素内容需要与某个指定元素的内容相同(eg:输入密码和重新输入密码)
                                                            --'#repsw' 为指定选择器
        }
      },
      
     messages : {                                              // -------错误信息需要根据元素和匹配规则指定
              email_name : {
                   required : '这是元素是必选项',
                   email : '必须满足邮箱的验证规则'             
              }
      },
      
      submitHandler : function () {             // 必须使用button进行表单提交操作,同时配合插件的submitHandler对提交操作进行处理
                console.log('呵呵呵');
      }
 });
    
 二  - 可以使用validate插件提供的方法自己添加指定规则
          - 调用$.validator.addMethod()添加规则
       - 参数1: 规则名称
       - 参数2: 规则的操作方式(函数)
                  - 形参1: 当前表单元素内容                                               (psw处输入的内容)
                  - 形参2: 当前表单元素(DOM对象)                                     (指定标签结构)
                  - 形参3: 使用规则时,设置的值(自定义规则通常不使用)        (rules中的 psw : true,中psw的标记,这里是true)
       - 参数3: 规则匹配失败时的提示信息

$.validator.addMethod('psw', function (value, element , params) {
      // console.log(value, element, params);

      console.log(this.optional(element));      //dependency-mismatch ---返回布尔值为true.
      
    return this.optional(element) || /^[a-z0-9A-Z_+-]{6,20}$/.test(value);

    }, '密码结构错误!~');



            - 使用optional的作用: 
                        - 必选框必须通过检测,
                        - 可选框可以不写,能够通过,但是如果填写内容,也必须通过检测
             -如果当前元素可选,并且没有输入内容: optional会返回这种结果: 'dependency-mismatch'   ---返回布尔值为true.
             -如果当前元素可选, 只要输入了内容,optional输出恒为false,

        -  equalTo    指定当前元素内容需要与某个指定元素的内容相同(eg:输入密码和重新输入密码)
