  1. 添加操作
发送ajax请求,从后台取回数据, 进行结构拼接
$.ajax({
    url : '/admin/api/_cate.php',
    success : function (datas) {
// 根据数据进行结构创建
    var str = '';
//对数据进行遍历
    $.each(datas, function (i, ele) {
        str += '<tr>\
        <td class="text-center"><input type="checkbox"></td>\
        <td>' + ele.name + '</td>\
        <td>' + ele.slug + '</td>\
        <td class="text-center">\
        <a href="javascript:;" class="btn btn-info btn-xs cate-edit" data-id="' + ele.id + '">编辑</a>\
        <a href="javascript:;" class="btn btn-danger btn-xs cate-del" data-id="' + ele.id + '">删除</a>\
        </td>\
    </tr>'
    });
    //拼接之后注意要添加到父元素里 
    $tbody.append(str);
    },
    //注意:进行结构创建时一定要将数据转换成js字符串的形式
    dataType : 'json'
});

           
 2. 删除操作
                    -  单条数据删除操作
//由于数据是动态创建的,所以此处需要使用事件委托, 将事件设置点击事件的父元素
$('#tbody').on('click', 'a.cate-del', function () {
// 向服务端页面_catedel.php发送请求，同时传递id值
    var id = $(this).data('id');
    var $tr = $(this).parents('tr');
$.ajax({
    url : '/admin/api/_catedel.php',
    data : {id : id},
    success : function (datas) {
// 判断服务端的数据删除操作是否成功，成功则删除页面结构
if (datas === '成功') {
    $tr.remove();
} else {
    alert('网络开小差了');
    }
    }
    });
});

            - 批量删除操作
                批量删除应先设置全选和反选功能,以及批量删除按钮的显示与隐藏,
                将要删除的数据的id保存在数组中, 点击删除时,将数组中的数据提交给服务端, 服务端从数据库中删除对应的数据, 前台只需对应删除结构即可
//全选反选
$selectAll.on('change', function () {
// 将全选按钮也设置为数据操作形式
// - 为了避免数据重复，可以先将idArr清空，再进行操作
idArr = [];

// - 设置底部的全选按钮状态与当前按钮状态相同（显示效果）
var $cbs = $('tbody input');
$cbs.prop('checked', $(this).prop('checked'))

// 数据处理操作：(全选按钮勾选的情况下进行)
// 一起设置批量删除按钮的显示隐藏效果
if ($(this).prop('checked')) {
$cbs.each(function (i, ele) {
idArr.push($(ele).data('id'));
});
$delsome.show();
} else {
$delsome.hide();
}
console.log(idArr);

});

// 3.2 底部复选框操作
// 设置一个数据，用于保存选中的选项对应的数据id
var idArr = [];
$tbody.on('change', 'input', function () {
// 原始的制作方式存在两个问题：
// - DOM操作较多 - 进行删除时的数据处理较为不便
// 3.2.1 根据当前选框是否选中，设置id数据的保存和移除
var id = $(this).data('id');
if ($(this).prop('checked')) {
// 如果选中，将id取出，保存在idArr中(需要在创建结构时给每个行的input设置data-id，方便操作)
idArr.push(id);
} else {
// 如果为取消勾选，从idArr中将当前数据对应的id移除
// - 注意：此处使用的indexOf具有兼容性问题，应当替换为遍历。
idArr.splice(idArr.indexOf(id), 1);
}
// 3.2.2 设置批量删除按钮和全选按钮的状态
// - 批量删除按钮的显示隐藏
idArr.length > 0 ? $delsome.show() : $delsome.hide();

// - 全选按钮的显示隐藏
var bool = $('tbody input').length === idArr.length;
$selectAll.prop('checked', bool);

console.log(idArr); 
});

批量删除操作
// - 点击批量删除，将idArr中的数据提交给服务端，服务端删除对应数据即可
$delsome.on('click', function () {
$.ajax({
    url : '/admin/api/_catedel.php',
    data : {id : idArr.join()},
    success : function (datas) {
// 检测，如果删除成功，需要进行结构等内容的处理操作
if (datas === '成功') {
// 从页面中找到选中的元素进行删除即可(可以不进行数据操作了)
    $('tbody input:checked').parents('tr').remove();
// 清空idArr
    idArr = [];
// 隐藏批量删除按钮
    $delsome.hide();
        } else {
        alert('网络开小差了，请稍后再试');
}
}
});
});

            3. 编辑操作
                    点击编辑时,应将当前点击条目的内容添加到对应的选框中,使用索引一一匹配即可
var $edit = $('#edit');
var $editCancel = $('#editCancel');

// - 设置表格中的编辑按钮点击功能
$tbody.on('click', 'a.cate-edit', function () {
// 将编辑的相关结构显示，将新增的按钮隐藏
    $edit.show();
    $editCancel.show();
    $add.hide();
// 点击后，将数据填写到左侧编辑的表单内，设置一个隐藏域用于保存id
var id = $(this).data('id');
var name = $(this).parents('tr').children('td').eq(1).text();
var slug = $(this).parents('tr').children('td').eq(2).text();

// 设置给表单
$form.find('input').eq(0).val(id);
$form.find('input').eq(1).val(name);
$form.find('input').eq(2).val(slug);
});

// 设置编辑按钮(提交功能)功能
$edit.on('click', function () {
// - 内容检测是否为空
    var datas = $form.serializeArray();

if (datas[0].value === '' || datas[1].value === '' || datas[2].value === '') {
    $errorBox.show();
    $errorBox.children('strong').text('内容没有填写完整');
    return;
}

// 如果成功，发送请求即可
$.ajax({
    url : '/admin/api/_cateedit.php',
    data : datas,
    success : function (datas) {
// 根据响应结果进行处理
    if (datas === '成功') {
        renderData();
        $reset.click();
        $edit.hide();
        $editCancel.hide();
        $add.show();
    } else {
        $errorBox.show();
        $errorBox.children('strong').text('网络开小差了，请稍后再试');
    }
    }
});
